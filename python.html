Q1: What are Python's built-in data types? Explain with examples.
Python has several built-in data types:

1. Numeric Types: - int: 10, -5, 0 - float: 3.14, 2.0, -0.5 - complex: 3+4j

2. Sequence Types: - str: "hello", 'python' - list: [1, 2, 3], ['a', 'b'] - tuple: (1, 2, 3), ('x', 'y')

3. Set Types: - set: {1, 2, 3}, {'a', 'b'} - frozenset: immutable set

4. Mapping Type: - dict: {'name': 'John', 'age': 25}

5. Boolean Type: - bool: True, False

6. Binary Types: - bytes: b'hello' - bytearray: mutable bytes - memoryview: memory view of objects 
# Examples of built-in data types
# Numeric
integer_num = 42
float_num = 3.14159
complex_num = 2 + 3j

# Sequence
string = "Hello, Python!"
list_data = [1, 2, 3, 'four', 5.0]
tuple_data = (1, 2, 'three')

# Set
set_data = {1, 2, 3, 3, 2}  # Duplicates removed: {1, 2, 3}
frozen_set = frozenset([1, 2, 3])

# Mapping
dictionary = {'name': 'Alice', 'age': 30, 'city': 'NYC'}

# Boolean
is_valid = True
is_complete = False

# Binary
bytes_data = b'binary data'
bytearray_data = bytearray(b' mutable bytes')

print(f"Type of integer_num: {type(integer_num)}")
print(f"Type of string: {type(string)}")
print(f"Type of dictionary: {type(dictionary)}")

Q2: Explain list comprehensions vs generator expressions
List Comprehensions: - Create a new list immediately - Use square brackets [] - Store all elements in memory - Good for smaller datasets

Generator Expressions: - Create a generator object - Use parentheses () - Generate elements on-the-fly (lazy evaluation) - Memory efficient for large datasets - Can only be iterated once 
# List Comprehension
numbers = [1, 2, 3, 4, 5]
squared_list = [x**2 for x in numbers]  # Creates list immediately
print(f"List comprehension: {squared_list}")
print(f"Type: {type(squared_list)}")
print(f"Memory: {squared_list} stores all values")

# Generator Expression
squared_gen = (x**2 for x in numbers)  # Creates generator
print(f"\nGenerator expression: {squared_gen}")
print(f"Type: {type(squared_gen)}")
print("Values are generated on demand:")

# Iterate through generator
for value in squared_gen:
    print(f"Generated: {value}")

# Generator is exhausted after first iteration
print(f"After iteration: {list(squared_gen)}")  # Empty

# Memory comparison
import sys
large_range = range(1000000)

# List comprehension uses more memory
list_mem = sys.getsizeof([x**2 for x in large_range])
gen_mem = sys.getsizeof((x**2 for x in large_range))

print(f"\nMemory usage for 1 million items:")
print(f"List comprehension: {list_mem / 1024 / 1024:.2f} MB")
print(f"Generator expression: {gen_mem / 1024:.2f} KB")

Q3: Explain *args and **kwargs in Python functions
*args (Non-keyword arguments): - Collects extra positional arguments as a tuple - Allows function to accept any number of positional arguments - The '*' unpacks the arguments

**kwargs (Keyword arguments): - Collects extra keyword arguments as a dictionary - Allows function to accept any number of keyword arguments - The '**' unpacks the keyword arguments

Order in function definition: 1. Standard positional arguments 2. *args 3. Standard keyword arguments 4. **kwargs 
def example_function(arg1, arg2, *args, kwarg1="default", **kwargs):
    """Demonstrates *args and **kwargs"""
    print(f"\nRequired arguments: {arg1}, {arg2}")
    print(f"Additional positional arguments (*args): {args}")
    print(f"Default keyword argument: {kwarg1}")
    print(f"Additional keyword arguments (**kwargs): {kwargs}")
    print(f"Type of args: {type(args)}")
    print(f"Type of kwargs: {type(kwargs)}")

# Different ways to call the function
print("=== Example 1: Basic usage ===")
example_function(1, 2)

print("\n=== Example 2: With extra positional args ===")
example_function(1, 2, 3, 4, 5)

print("\n=== Example 3: With keyword args ===")
example_function(1, 2, kwarg1="custom", extra1="value1", extra2="value2")

print("\n=== Example 4: Unpacking lists and dicts ===")
numbers = [10, 20, 30, 40, 50]
config = {'color': 'blue', 'size': 'large', 'speed': 'fast'}

example_function(*numbers, **config)

Q4: Explain Python's Method Resolution Order (MRO) in multiple inheritance
Method Resolution Order (MRO): - Determines the order in which base classes are searched when looking for a method - Uses C3 linearization algorithm - Ensures monotonicity (consistent ordering)

Key Principles: 1. Children before parents 2. Order of inheritance matters 3. Left to right in class definition

Viewing MRO: - Use ClassName.__mro__ - Use ClassName.mro() 
class A:
    def method(self):
        return "A.method"

class B(A):
    def method(self):
        return "B.method"

class C(A):
    def method(self):
        return "C.method"

class D(B, C):
    pass

class E(C, B):
    pass

# Diamond inheritance problem
print("=== Diamond Inheritance Example ===")
d = D()
print(f"D().method(): {d.method()}")
print(f"D MRO: {[cls.__name__ for cls in D.__mro__]}")

e = E()
print(f"\nE().method(): {e.method()}")
print(f"E MRO: {[cls.__name__ for cls in E.__mro__]}")

# Complex inheritance example
class F:
    def process(self):
        return "F.process"

class G(F):
    def process(self):
        return "G.process"

class H(F):
    def process(self):
        return "H.process"

class I(G, H):
    pass

class J(H, G):
    pass

print("\n=== Complex Inheritance ===")
i = I()
print(f"I().process(): {i.process()}")
print(f"I MRO: {[cls.__name__ for cls in I.__mro__]}")

j = J()
print(f"\nJ().process(): {j.process()}")
print(f"J MRO: {[cls.__name__ for cls in J.__mro__]}")

Q5: Compare @classmethod, @staticmethod, and instance methods
Instance Methods: - First parameter: self (instance reference) - Can access and modify instance attributes - Can access class attributes via self.__class__

@classmethod: - First parameter: cls (class reference) - Can't modify instance state (no self) - Can modify class state (affects all instances) - Often used as alternative constructors

@staticmethod: - No special first parameter (no self, no cls) - Can't modify instance or class state - Just a regular function inside a class - Used for utility/helper functions 
class Employee:
    # Class variable
    company = "TechCorp"
    total_employees = 0
    
    def __init__(self, name, salary):
        self.name = name
        self.salary = salary
        Employee.total_employees += 1
    
    # Instance method - works with instance data
    def get_details(self):
        return f"{self.name} - ${self.salary}/year"
    
    def give_raise(self, percentage):
        self.salary *= (1 + percentage/100)
        return self.salary
    
    # Class method - works with class data
    @classmethod
    def create_from_string(cls, emp_string):
        """Alternative constructor: 'John Doe,50000'"""
        name, salary = emp_string.split(',')
        return cls(name.strip(), float(salary))
    
    @classmethod
    def get_company_info(cls):
        return f"{cls.company} has {cls.total_employees} employees"
    
    @classmethod
    def change_company(cls, new_name):
        cls.company = new_name
    
    # Static method - utility function
    @staticmethod
    def calculate_bonus(salary, performance_rating):
        """Calculate bonus based on salary and performance"""
        if performance_rating == 'A':
            return salary * 0.20
        elif performance_rating == 'B':
            return salary * 0.10
        else:
            return salary * 0.05
    
    @staticmethod
    def is_valid_salary(salary):
        """Validate salary amount"""
        return salary > 0 and salary < 1000000

# Usage examples
print("=== Instance Methods ===")
emp1 = Employee("Alice", 60000)
print(f"Details: {emp1.get_details()}")
emp1.give_raise(10)
print(f"After raise: {emp1.get_details()}")

print("\n=== Class Methods ===")
emp2 = Employee.create_from_string("Bob Smith, 75000")
print(f"Created from string: {emp2.get_details()}")
print(f"Company info: {Employee.get_company_info()}")

# Change company name for all employees
Employee.change_company("MetaTech")
print(f"After change: {Employee.company}")
print(f"emp1 company: {emp1.company}")
print(f"emp2 company: {emp2.company}")

print("\n=== Static Methods ===")
print(f"Bonus for $60000 with rating 'A': ${Employee.calculate_bonus(60000, 'A'):,.2f}")
print(f"Is $500000 valid salary? {Employee.is_valid_salary(500000)}")
print(f"Is $-1000 valid salary? {Employee.is_valid_salary(-1000)}")

# Practical example: Database Model
class DatabaseModel:
    _connection = None
    
    @classmethod
    def connect(cls, connection_string):
        """Establish database connection for all instances"""
        cls._connection = f"Connected to {connection_string}"
        return cls._connection
    
    @classmethod
    def get_connection(cls):
        return cls._connection
    
    @staticmethod
    def validate_query(query):
        """Validate SQL query syntax"""
        if not query or not isinstance(query, str):
            return False
        return query.strip().upper().startswith(('SELECT', 'INSERT', 'UPDATE', 'DELETE'))

print("\n=== Database Model Example ===")
DatabaseModel.connect("postgresql://localhost/mydb")
print(f"Connection: {DatabaseModel.get_connection()}")
print(f"Valid query 'SELECT * FROM users': {DatabaseModel.validate_query('SELECT * FROM users')}")
print(f"Valid query 'invalid query': {DatabaseModel.validate_query('invalid query')}")
